cmake_minimum_required(VERSION 3.15)
project(NForge LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Options
option(BUILD_WITH_CUDA "Build with CUDA support" OFF)


# Output dirs
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)


# Sources
set(NFORGE_SRC
	src/tensor.cpp
)

if(BUILD_WITH_CUDA)
	enable_language(CUDA)
	list(APPEND NFORGE_SRC src/tensor_cuda.cu)
endif()

add_library(nforge STATIC ${NFORGE_SRC})

# Include headers
target_include_directories(nforge PUBLIC
	$<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
)


if(BUILD_WITH_CUDA)
	target_compile_definitions(nforge PUBLIC NFORGE_USE_CUDA)
endif()


# Enable testing
enable_testing()
add_subdirectory(tests)
